What's missing here?

input validation! 
              
                  apiVersion: apiextensions.k8s.io/v1beta1
                  kind: CustomResourceDefinition
                  metadata:
                    name: openvpns.ptlc.nl
                  spec:
                    group: ptlc.nl
                    version: v1alpha1
                    scope: Namespaced
                    names:
                      kind: OpenVPN
                      listKind: OpenVPNList
                      plural: openvpns
                      singular: openvpn
                    validation:
                      openAPIV3Schema:
                        properties:
                          spec:
                            properties:
                              pidCIDR
                                type: string
                                pattern: ^(([0-9]|[1-9][0-9]|1[0-9]{2}|2[0-4][0-9]|25[0-5])\.){3}([0-9]|[1-9][0-9]|1[0-9]{2}|2[0-4][0-9]|25[0-5])(\/([0-9]|[1-2][0-9]|3[0-2]))$
                              serviceCIDR:
                                type: string
                                pattern: ^(([0-9]|[1-9][0-9]|1[0-9]{2}|2[0-4][0-9]|25[0-5])\.){3}([0-9]|[1-9][0-9]|1[0-9]{2}|2[0-4][0-9]|25[0-5])(\/([0-9]|[1-2][0-9]|3[0-2]))$
